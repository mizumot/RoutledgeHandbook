First, you should check the following assumptions.¶

  1. Whether data is normally distributed.
  2. Whether groups have similar variances.

Drawing beeswarm plot¶

In [6]:
# Drawing boxplots
boxplot(X, Y, xlab="Group", ylab="Score", las=1) # las=1 produces labels which are horizontally oriented.

# Adding individual data points to the boxplot
library(beeswarm) # Load the beeswarm package
beeswarm(score ~ group, data=dat, col=1, pch=16, add=TRUE)

# Adding arrows that show means and +/-1 SDs
X <- subset(dat, group == "X") # Include only Group X
points(1.2, mean(X$score), pch=18, col=1, cex=2)
arrows(1.2, mean(X$score), 1.2, mean(X$score)+sd(X$score), length=0.1, angle=45, col=1)
arrows(1.2, mean(X$score), 1.2, mean(X$score)-sd(X$score), length=0.1, angle=45, col=1)
Y <- subset(dat, group == "Y") # Include only Group Y
points(2.2, mean(Y$score), pch=18, col=1, cex=2)
arrows(2.2, mean(Y$score), 2.2, mean(Y$score)+sd(Y$score), length=0.1, angle=45, col=1)
arrows(2.2, mean(Y$score), 2.2, mean(Y$score)-sd(Y$score), length=0.1, angle=45, col=1)
